/*-- QUESTIONS RELATED TO CUSTOMERS
     [Q1] What is the distribution of customers across states?
     Hint: For each state, count the number of customers.*/

SELECT 
    state, COUNT(*) Count_Customers
FROM
    customer_t
GROUP BY state
ORDER BY state;

/* [Q2] What is the average rating in each quarter?
-- Very Bad is 1, Bad is 2, Okay is 3, Good is 4, Very Good is 5.

Hint: Use a common table expression and in that CTE, assign numbers to the different customer ratings. 
      Now average the feedback for each quarter. 

Note: For reference, refer to question number 4. Week-2: mls_week-2_gl-beats_solution-1.sql. 
      You'll get an overview of how to use common table expressions from this question.*/


 with Avg_rating as (
   select customer_id,
         quarter_number,
         customer_feedback,
         case when customer_feedback ='Very Bad' then 1
              when customer_feedback ='Okay' then 3
              when customer_feedback ='Bad' then 2
              when customer_feedback ='Good' then 4
              else
              5
         end as rating
   from order_t 
   )
   select 
   quarter_number Qtr,
   avg(rating) Avg_Rating
   from Avg_rating
   group by 1
   order by 1 ;
   

-- ---------------------------------------------------------------------------------------------------------------------------------

/* [Q3] Are customers getting more dissatisfied over time?

Hint: Need the percentage of different types of customer feedback in each quarter. Use a common table expression and
	  determine the number of customer feedback in each category as well as the total number of customer feedback in each quarter.
	  Now use that common table expression to find out the percentage of different types of customer feedback in each quarter.
      Eg: (total number of very good feedback/total customer feedback)* 100 gives you the percentage of very good feedback.
      
Note: For reference, refer to question number 4. Week-2: mls_week-2_gl-beats_solution-1.sql. 
      You'll get an overview of how to use common table expressions from this question.*/


      
with CTE1 as
(
select quarter_number,
 sum(case when customer_feedback ='Very Bad' then 1 else 0 end) as Very_Bad,
 sum(case when customer_feedback ='Okay' then 1 else 0 end) as Okay,
 sum(Case when customer_feedback ='Bad' then 1 else 0 end) as Bad,
 sum(case when customer_feedback ='Good' then 1 else 0 end) as Good,
 sum(case when customer_feedback ='Very Good' then 1 else 0 end) as Very_Good,
 count(customer_feedback) as total_feedbacks
 from order_t
group by 1
)
select quarter_number Qtr,
round((Very_bad/total_feedbacks)*100,2) as Very_Bad_Perc,
round((Okay/total_feedbacks)*100,2) as Okay_Perc,
round((Bad/total_feedbacks)*100,2) as Bad_Perc,
round((Good/total_feedbacks)*100,2) as Good_Perc,
round((Very_Good/total_feedbacks)*100,2) as Very_Good_Perc
from CTE1
order by 1

/*[Q4] Which are the top 5 vehicle makers preferred by the customer.

Hint: For each vehicle make what is the count of the customers.*/


SELECT 
    prod.vehicle_maker AS Vehicle_make, COUNT(ordr.customer_id) AS Number_Of_Customers
FROM
    order_t ordr
        JOIN
    product_t prod USING (product_id)
GROUP BY prod.vehicle_maker
ORDER BY 2 desc
LIMIT 5;


-- ---------------------------------------------------------------------------------------------------------------------------------

/*[Q5] What is the most preferred vehicle make in each state?

Hint: Use the window function RANK() to rank based on the count of customers for each state and vehicle maker. 
After ranking, take the vehicle maker whose rank is 1.*/

with CTE1 as
(
 select cust.state,
        prod.vehicle_maker ,
        prod.vehicle_model,
        count(* ) Cnt_Customer,
       rank() over(partition by cust.state order by prod.vehicle_maker) AS RNK
 from order_t ord  join customer_t cust using (customer_id)
       join product_t prod using (product_id)
       group by cust.state,prod.vehicle_maker,prod.vehicle_model
       )
       select state,vehicle_maker ,vehicle_model from CTE1 where rnk =1
       order by 1;

/* [Q7] What is the quarter over quarter % change in revenue? 

Hint: Quarter over Quarter percentage change in revenue means what is the change in revenue from the subsequent quarter to the previous quarter in percentage.
      To calculate you need to use the common table expression to find out the sum of revenue for each quarter.
      Then use that CTE along with the LAG function to calculate the QoQ percentage change in revenue.
*/


with QoQ as
 (
 SELECT
          ord.quarter_number,
          SUM(ord.quantity*(ord.vehicle_price - ((ord.discount/100)*ord.vehicle_price))) Revenue
    FROM order_t ord
	INNER JOIN product_t prod
	    ON ord.product_id = prod.product_id
	GROUP BY  1
    )
    SELECT
      quarter_number,
      revenue,
      LAG(revenue) OVER(ORDER BY quarter_number) AS previous_revenue,
      (revenue - LAG(revenue) OVER(ORDER BY quarter_number))/LAG(revenue) OVER(ORDER BY quarter_number) AS QoQ_perc_change
FROM QoQ;

          SUM(ord.quantity*(ord.vehicle_price - ((ord.discount/100)*ord.vehicle_price))) Revenue,
          count(*) Num_of_orders
    FROM order_t ord
	INNER JOIN product_t prod
	    ON ord.product_id = prod.product_id
	GROUP BY  1
    order by 1

/* QUESTIONS RELATED TO SHIPPING 
    [Q9] What is the average discount offered for different types of credit cards?

Hint: Find out the average of discount for each credit card type.*/

SELECT 
    cust.credit_card_type Credit_card, AVG(ord.discount) Avg_Discount
FROM
    customer_t cust
        JOIN
    order_t ord USING (customer_id)
GROUP BY credit_card_type
ORDER BY 1

/* [Q10] What is the average time taken to ship the placed orders for each quarters?
	Hint: Use the dateiff function to find the difference between the ship date and the order date.
*/


SELECT quarter_number,
    AVG(DATEDIFF(ord.ship_date, ord.order_date)) Avg_time
FROM
    order_t ord join shipper_t ship using(shipper_id)
GROUP BY  quarter_number
order by quarter_number
